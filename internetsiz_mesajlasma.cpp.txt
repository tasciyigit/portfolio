int pinIndex(int pin){
  for(int i=0;i<BTN_COUNT;i++) if(BTN_PINS[i]==pin) return i;
  return -1;
}

bool buttonPressedPin(int pin){
  int idx = pinIndex(pin);
  if(idx < 0) return false;
  if(digitalRead(pin) == LOW){
    unsigned long now = millis();
    if(now - lastPressTime[idx] > DEBOUNCE_MS){
      lastPressTime[idx] = now;
      return true;
    }
  }
  return false;
}

// LED durumu: bağlı / bağlı değil
void setConnectedLED(bool connected) {
  if (connected) {
    digitalWrite(PIN_LED_GREEN, HIGH);
    digitalWrite(PIN_LED_RED, LOW);
  } else {
    digitalWrite(PIN_LED_GREEN, LOW);
    digitalWrite(PIN_LED_RED, HIGH);
  }
}

// Buzzer bip isteği (seçili pattern ile)
void requestBeep() {
  if (!buzzerEnabled) return;
  if (buzzerPlaying) return;

  buzzerPlaying = true;
  buzzerStateOn = true;
  buzzerLastChange = millis();

  switch (buzzerPattern) {
    case 0: buzzerRemainingBeeps = 1; break;
    case 1: buzzerRemainingBeeps = 2; break;
    case 2: buzzerRemainingBeeps = 3; break;
    default: buzzerRemainingBeeps = 1; break;
  }

  digitalWrite(PIN_BUZZER, HIGH);
}

void handleBuzzer() {
  if (!buzzerPlaying) return;

  unsigned long now = millis();
  if (buzzerStateOn) {
    if (now - buzzerLastChange >= BEEP_LEN_MS) {
      buzzerStateOn = false;
      buzzerLastChange = now;
      digitalWrite(PIN_BUZZER, LOW);
      buzzerRemainingBeeps--;
      if (buzzerRemainingBeeps <= 0) {
        buzzerPlaying = false;
      }
    }
  } else {
    if (buzzerRemainingBeeps > 0 && (now - buzzerLastChange >= BEEP_GAP_MS)) {
      buzzerStateOn = true;
      buzzerLastChange = now;
      digitalWrite(PIN_BUZZER, HIGH);
    }
  }
}

void addMessageToHistory(const String &msg, bool incoming, const String &sender) {
  if (msg.length() == 0) return;

  String m = msg;
  if (m.length() > 80) m = m.substring(0, 80);

  historyBuf[historyHead].text = m;
  historyBuf[historyHead].incoming = incoming;
  historyBuf[historyHead].sender = sender;

  historyHead = (historyHead + 1) % MAX_HISTORY;
  if (historyCount < MAX_HISTORY) historyCount++;

  if (currentMode == MODE_LOG && logOffset == 0) {
    drawUI();
  }
}

// Türkçe karakterleri ASCII'ye çevir
String normalizeTurkish(const String &s) {
  String out;
  int len = s.length();
  for (int i = 0; i < len; i++) {
    unsigned char c = (unsigned char)s[i];

    if (c < 0x80) {
      out += (char)c;
    } else if (c == 0xC3 && i + 1 < len) {
      unsigned char c2 = (unsigned char)s[i+1];
      if      (c2 == 0xA7) { out += 'c'; i++; }
      else if (c2 == 0x87) { out += 'C'; i++; }
      else if (c2 == 0xB6) { out += 'o'; i++; }
      else if (c2 == 0x96) { out += 'O'; i++; }
      else if (c2 == 0xBC) { out += 'u'; i++; }
      else if (c2 == 0x9C) { out += 'U'; i++; }
      else { i++; }
    } else if (c == 0xC4 && i + 1 < len) {
      unsigned char c2 = (unsigned char)s[i+1];
      if      (c2 == 0x9F) { out += 'g'; i++; }
      else if (c2 == 0x9E) { out += 'G'; i++; }
      else if (c2 == 0xB1) { out += 'i'; i++; }
      else if (c2 == 0xB0) { out += 'I'; i++; }
      else { i++; }
    } else if (c == 0xC5 && i + 1 < len) {
      unsigned char c2 = (unsigned char)s[i+1];
      if      (c2 == 0x9F) { out += 's'; i++; }
      else if (c2 == 0x9E) { out += 'S'; i++; }
      else { i++; }
    } else {
      // Diğerleri atlanır
    }
  }
  return out;
}

void handleIncomingMessage(const String &payload) {
  incomingMsg = payload;

  String sender = "Bilinmiyor";
  String text   = payload;

  if (payload.startsWith("N:")) {
    int sep = payload.indexOf("|M:");
    if (sep > 2) {
      sender = payload.substring(2, sep);
      text   = payload.substring(sep + 3);
    }
  }

  sender = normalizeTurkish(sender);
  text   = normalizeTurkish(text);

  addMessageToHistory(text, true, sender);

  if (currentMode != MODE_LOG) {
    requestBeep();
  }
}

bool isLeapYear(int y) {
  return ((y % 4 == 0) && (y % 100 != 0)) || (y % 400 == 0);
}

int daysInMonth(int y, int m) {
  static const int days[12] = {31,28,31,30,31,30,31,31,30,31,30,31};
  if (m < 1 || m > 12) return 31;
  if (m == 2 && isLeapYear(y)) return 29;
  return days[m-1];
}

void updateClock() {
  unsigned long now = millis();
  if (now - lastClockUpdate >= 1000) {
    lastClockUpdate += 1000;

    clockSecond++;
    if (clockSecond >= 60) {
      clockSecond = 0;
      clockMinute++;
      if (clockMinute >= 60) {
        clockMinute = 0;
        clockHour++;
        if (clockHour >= 24) {
          clockHour = 0;
          if (dateValid) {
            clockDay++;
            int dim = daysInMonth(clockYear, clockMonth);
            if (clockDay > dim) {
              clockDay = 1;
              clockMonth++;
              if (clockMonth > 12) {
                clockMonth = 1;
                clockYear++;
              }
            }
          }
        }
      }
    }

    if (currentMode == MODE_CLOCK) {
      drawUI();
    }
  }
}

void setTimeFromMessage(const String &msg) {
  if (!msg.startsWith("TIME:")) return;
  String t = msg.substring(5);

  int hh = 0, mm = 0, ss = 0;

  if (t.length() >= 19 && t.charAt(4) == '-' && t.charAt(7) == '-' && t.charAt(10) == ' ') {
    int yy = t.substring(0,4).toInt();
    int mo = t.substring(5,7).toInt();
    int dd = t.substring(8,10).toInt();
    hh = t.substring(11,13).toInt();
    mm = t.substring(14,16).toInt();
    ss = t.substring(17,19).toInt();

    if (hh < 0 || hh > 23) return;
    if (mm < 0 || mm > 59) return;
    if (ss < 0 || ss > 59) return;

    if (yy >= 2000 && mo >= 1 && mo <= 12 && dd >= 1 && dd <= 31) {
      clockYear = yy;
      clockMonth = mo;
      clockDay = dd;
      dateValid = true;
      Serial.printf("Date synced: %04d-%02d-%02d\n", yy, mo, dd);
    }
  } else if (t.length() >= 8) {
    hh = t.substring(0,2).toInt();
    mm = t.substring(3,5).toInt();
    ss = t.substring(6,8).toInt();

    if (hh < 0 || hh > 23) return;
    if (mm < 0 || mm > 59) return;
    if (ss < 0 || ss > 59) return;
  } else {
    return;
  }

  clockHour   = hh;
  clockMinute = mm;
  clockSecond = ss;
  lastClockUpdate = millis();

  Serial.printf("Time synced: %02d:%02d:%02d\n", hh, mm, ss);
}

bool snakeOccupies(int x, int y) {
  for (int i = 0; i < snakeLength; i++) {
    if (snakeX[i] == x && snakeY[i] == y) return true;
  }
  return false;
}

void placeFood() {
  for (int tries = 0; tries < 100; tries++) {
    int fx = random(GAME_COLS);
    int fy = random(GAME_ROWS);
    if (!snakeOccupies(fx, fy)) {
      foodX = fx;
      foodY = fy;
      return;
    }
  }
  foodX = 0;
  foodY = 0;
}

void resetGame() {
  snakeLength = 3;
  int startX = GAME_COLS / 2;
  int startY = GAME_ROWS / 2;

  snakeX[0] = startX;
  snakeY[0] = startY;
  snakeX[1] = startX - 1;
  snakeY[1] = startY;
  snakeX[2] = startX - 2;
  snakeY[2] = startY;

  snakeDirX = 1;
  snakeDirY = 0;

  gameRunning = true;
  gameOver = false;
  lastGameUpdate = millis();

  placeFood();
  drawUI();
}

void updateGame() {
  if (currentMode != MODE_GAME) return;
  if (!gameRunning) return;

  unsigned long now = millis();
  if (now - lastGameUpdate < GAME_SPEED_MS) return;
  lastGameUpdate = now;

  int newHeadX = snakeX[0] + snakeDirX;
  int newHeadY = snakeY[0] + snakeDirY;

  if (newHeadX < 0 || newHeadX >= GAME_COLS || newHeadY < 0 || newHeadY >= GAME_ROWS) {
    gameOver = true;
    gameRunning = false;
    requestBeep();
    drawUI();
    return;
  }

  for (int i = 0; i < snakeLength; i++) {
    if (snakeX[i] == newHeadX && snakeY[i] == newHeadY) {
      gameOver = true;
      gameRunning = false;
      requestBeep();
      drawUI();
      return;
    }
  }

  int tailX = snakeX[snakeLength - 1];
  int tailY = snakeY[snakeLength - 1];

  for (int i = snakeLength - 1; i > 0; i--) {
    snakeX[i] = snakeX[i - 1];
    snakeY[i] = snakeY[i - 1];
  }

  snakeX[0] = newHeadX;
  snakeY[0] = newHeadY;

  if (newHeadX == foodX && newHeadY == foodY) {
    if (snakeLength < GAME_MAX_LENGTH) {
      snakeLength++;
      snakeX[snakeLength - 1] = tailX;
      snakeY[snakeLength - 1] = tailY;
    }
    placeFood();
  }

  drawUI();
}

void handleRoot() {
  server.send_P(200, "text/html", HTML_page);
}

void handleSend() {
  if (server.hasArg("msg")) {
    String m = server.arg("msg");
    if (m.length()>400) m = m.substring(0,400);

    handleIncomingMessage(m);
    webSocket.broadcastTXT(m);
    server.send(200, "text/plain", "OK");
    Serial.printf("HTTP /send received: %s\n", m.c_str());
  } else {
    server.send(400, "text/plain", "No 'msg' provided");
  }
}

// ---------------------------------------------------------------------------
// WebSocket event
// ---------------------------------------------------------------------------
void webSocketEvent(uint8_t num, WStype_t type, uint8_t * payload, size_t length) {
  switch(type) {
    case WStype_CONNECTED: {
      IPAddress ip = webSocket.remoteIP(num);
      Serial.printf("WS client %u connected from %d.%d.%d.%d\n",
                    num, ip[0],ip[1],ip[2],ip[3]);
      anyClientConnected = true;
      setConnectedLED(true);

      if(incomingMsg.length()>0) webSocket.sendTXT(num, incomingMsg);
      break;
    }
    case WStype_DISCONNECTED: {
      Serial.printf("WS client %u disconnected\n", num);
      anyClientConnected = false;
      setConnectedLED(false);
      break;
    }
    case WStype_TEXT: {
      String msg = String((char*)payload);

      if (msg.startsWith("TIME:")) {
        setTimeFromMessage(msg);
        break;
      }

      Serial.printf("WS msg from client %u: %s\n", num, msg.c_str());
      handleIncomingMessage(msg);
      webSocket.broadcastTXT(msg);
      break;
    }
    default:
      break;
  }
}

    line1 = txt;
    line2 = "";
  } else if (txt.length() <= 2*maxCharsPerLine) {
    line1 = txt.substring(0, maxCharsPerLine);
    line2 = txt.substring(maxCharsPerLine);

    line2 = last.substring(maxCharsPerLine);

void showSplashScreen() {
  display.clearDisplay();
  display.setTextColor(SSD1306_WHITE);
  display.setTextSize(1);

  display.setCursor(10, 10);
  display.print("Hosgeldin Yigit");
  display.setCursor(10, 24);
  display.print("Hazirlaniyor...");

  int barX = 10;
  int barY = 40;
  int barW = 108;
  int barH = 10;
  display.drawRect(barX, barY, barW, barH, SSD1306_WHITE);
  display.display();

  unsigned long start = millis();
  const unsigned long duration = 5000;

  while (millis() - start < duration) {
    unsigned long elapsed = millis() - start;
    float ratio = (float)elapsed / (float)duration;
    if (ratio > 1.0f) ratio = 1.0f;

    int fillW = (int)((barW - 2) * ratio);

    display.fillRect(barX + 1, barY + 1, barW - 2, barH - 2, SSD1306_BLACK);
    display.fillRect(barX + 1, barY + 1, fillW, barH - 2, SSD1306_WHITE);

    display.display();
    delay(80);
  }

  delay(200);
}

void setupPinsAndDisplay() {
  for (int i=0;i<BTN_COUNT;i++) {
    pinMode(BTN_PINS[i], INPUT_PULLUP);
  }

  pinMode(PIN_LED_GREEN, OUTPUT);
  pinMode(PIN_LED_RED, OUTPUT);
  pinMode(PIN_BUZZER, OUTPUT);

  digitalWrite(PIN_LED_GREEN, LOW);
  digitalWrite(PIN_LED_RED, HIGH);
  digitalWrite(PIN_BUZZER, LOW);

  Wire.begin(21,22);
  if (!display.begin(SSD1306_SWITCHCAPVCC, 0x3C)) {
    Serial.println("SSD1306 init failed");
    while(true) delay(1000);
  }
  display.clearDisplay();
  applyBrightness();
  display.display();
}

  applyWifiPower();   // <<< seçilen moda göre WiFi gücünü ayarla

  Serial.print("AP IP: ");
  Serial.println(WiFi.softAPIP());

  server.on("/", handleRoot);
  server.on("/send", HTTP_POST, handleSend);
  server.begin();
  Serial.println("HTTP server started");

  webSocket.begin();
  webSocket.onEvent(webSocketEvent);
  Serial.println("WebSocket server started on port 81");

  typedMsg = "";
  incomingMsg = "";

  currentMode = MODE_COMPOSE;
  logOffset = 0;
  settingsIndex = 0;
  settingsTopIndex = 0;
  modeSelectIndex = (int)currentMode;
  logDetailMode = false;

  setConnectedLED(false);

  clockHour = 12;
  clockMinute = 0;
  clockSecond = 0;
  lastClockUpdate = millis();
  dateValid = false;

  drawUI();
}

  updateClock();
  updateGame();

  bool upPressed    = buttonPressedPin(PIN_UP);
  bool downPressed  = buttonPressedPin(PIN_DOWN);
  bool leftPressed  = buttonPressedPin(PIN_LEFT);
  bool rightPressed = buttonPressedPin(PIN_RIGHT);
  bool modePressed  = buttonPressedPin(PIN_MODE);

  if (modePressed) {
    if (currentMode != MODE_SELECT) {
      modeSelectIndex = (int)currentMode;
      if (modeSelectIndex >= MODE_COUNT) modeSelectIndex = 0;
      currentMode = MODE_SELECT;
      drawUI();
    } else {
      currentMode = MODE_COMPOSE;
      drawUI();
    }
  }

  switch(currentMode) {
        case MODE_COMPOSE:
      if (upPressed) {
        // En üst satırdaysak yukarı basınca en alta git
        cy = (cy == 0) ? (ROWS - 1) : (cy - 1);
        drawUI();
      }
      else if (downPressed) {
        // En alt satırdaysak aşağı basınca en üste git
        cy = (cy == ROWS - 1) ? 0 : (cy + 1);
        drawUI();
      }
      else if (leftPressed) {
        // En soldaysak sola basınca en sağa git
        cx = (cx == 0) ? (COLS - 1) : (cx - 1);
        drawUI();
      }
      else if (rightPressed) {
        // En sağdaysak sağa basınca en sola git
        cx = (cx == COLS - 1) ? 0 : (cx + 1);
        drawUI();
      }
      break;


    case MODE_LOG:
      if (upPressed && historyCount > 0 && logOffset < historyCount - 1) {
        logOffset++;
        drawUI();
      } else if (downPressed && logOffset > 0) {
        logOffset--;
        drawUI();
      }

      if (rightPressed && historyCount > 0) {
        logDetailMode = true;
        drawUI();
      } else if (leftPressed && logDetailMode) {
        logDetailMode = false;
        drawUI();
      }
      break;

        case MODE_SETTINGS: {
      const int visibleLines = 5;

      if (upPressed) {
        if (settingsIndex == 0) {
          // En üstteyken yukarı basınca en alta sar
          settingsIndex = SETTINGS_COUNT - 1;
          settingsTopIndex = max(0, SETTINGS_COUNT - visibleLines);
        } else {
          settingsIndex--;
          if (settingsIndex < settingsTopIndex) {
            settingsTopIndex = settingsIndex;
          }
        }
        drawUI();
      }
      else if (downPressed) {
        if (settingsIndex == SETTINGS_COUNT - 1) {
          // En alttayken aşağı basınca en üste sar
          settingsIndex = 0;
          settingsTopIndex = 0;
        } else {
          settingsIndex++;
          if (settingsIndex > settingsTopIndex + visibleLines - 1) {
            settingsTopIndex = settingsIndex - (visibleLines - 1);
          }
        }
        drawUI();
      }

      // alttaki left/right + changed/saveSettings kısmı aynen devam


      if (leftPressed || rightPressed) {
        bool changed = false;

        if (settingsIndex == 0) {         // Ses
          buzzerEnabled = !buzzerEnabled;
          changed = true;
          drawUI();
        } else if (settingsIndex == 1) {  // Zil
          int delta = rightPressed ? 1 : -1;
          buzzerPattern += delta;
          if (buzzerPattern < 0) buzzerPattern = 2;
          if (buzzerPattern > 2) buzzerPattern = 0;
          changed = true;
          requestBeep();
          drawUI();
        } else if (settingsIndex == 2) {  // Parlaklık
          if (leftPressed && brightnessLevel > 0) {
            brightnessLevel--;
            changed = true;
          } else if (rightPressed && brightnessLevel < 2) {
            brightnessLevel++;
            changed = true;
          }
          applyBrightness();
          drawUI();
        } else if (settingsIndex == 3) {  // WiFi Gücü
          wifiPowerMode = 1 - wifiPowerMode;
          applyWifiPower();
          changed = true;
          drawUI();
        } else if (settingsIndex == 4) {  // Dil
          languageMode = 1 - languageMode;
          changed = true;
          drawUI();
        }

        if (changed) {
          saveSettings();
        }
      }
      break;
    }

    case MODE_CLOCK:
      break;

    case MODE_GAME:
      if (upPressed && snakeDirY != 1) {
        snakeDirX = 0;
        snakeDirY = -1;
      } else if (downPressed && snakeDirY != -1) {
        snakeDirX = 0;
        snakeDirY = 1;
      } else if (leftPressed && snakeDirX != 1) {
        snakeDirX = -1;
        snakeDirY = 0;
      } else if (rightPressed && snakeDirX != -1) {
        snakeDirX = 1;
        snakeDirY = 0;
      }

      if (buttonPressedPin(PIN_OK) && gameOver) {
        resetGame();
      }
      break;

        case MODE_SELECT:
      if (upPressed) {
        // En üstten yukarıya basınca en alta sar
        if (modeSelectIndex == 0) {
          modeSelectIndex = MODE_COUNT - 1;
        } else {
          modeSelectIndex--;
        }
        drawUI();
      }
      else if (downPressed) {
        // En alttan aşağıya basınca en üste sar
        if (modeSelectIndex == MODE_COUNT - 1) {
          modeSelectIndex = 0;
        } else {
          modeSelectIndex++;
        }
        drawUI();
      }

      if (buttonPressedPin(PIN_OK)) {
        currentMode = (UIMode)modeSelectIndex;
        if (currentMode == MODE_LOG) {
          logOffset = 0;
          logDetailMode = false;
        } else if (currentMode == MODE_GAME) {
          resetGame();
        }
        drawUI();
      }
      break;

  }

  if (currentMode == MODE_COMPOSE) {
    int okIdx = pinIndex(PIN_OK);
    int okRaw = digitalRead(PIN_OK);

    if (okIdx >= 0) {
      if (okRaw == LOW && !okDown) {
        unsigned long now = millis();
        if (now - lastPressTime[okIdx] > DEBOUNCE_MS) {
          okDown = true;
          okDownTime = now;
          lastPressTime[okIdx] = now;
        }
      }

      if (okRaw == HIGH && okDown) {
        unsigned long dur = millis() - okDownTime;
        char k = useAltKeyboard ? altKeys[cy][cx] : keys[cy][cx];

        if (k == '<') {
          if (dur < OK_HOLD_MS) {
            if (typedMsg.length() > 0) typedMsg.remove(typedMsg.length()-1);
          } else {
            if (typedMsg.length() < 200) typedMsg += ' ';
          }
        } else if (k == '@') {
          if (dur < OK_HOLD_MS) {
            String msgText = typedMsg.length() ? typedMsg : "(empty)";
            String payload = String("N:") + DEVICE_NAME + "|M:" + msgText;

            webSocket.broadcastTXT(payload);

            String dispText = normalizeTurkish(msgText);
            addMessageToHistory(dispText, false, normalizeTurkish(DEVICE_NAME));

            Serial.printf("Device broadcast: %s\n", msgText.c_str());
            typedMsg = "";
          } else {
            useAltKeyboard = !useAltKeyboard;
          }
        } else {
          char ch = k;
          if (isalpha((unsigned char)k)) {
            if (dur < OK_HOLD_MS) {
              ch = (char)tolower((unsigned char)k);
            } else {
              ch = k;
            }
          }
          if (typedMsg.length() < 200) {
            typedMsg += ch;
          }
        }

        okDown = false;
        drawUI();
      }
    }
  }

  delay(20);
}
